MATCH (:User {facebook_id: '${senderID}'})-[:Owns]->(e:Event {scheduled: FALSE})-[:Reminds]->(u:User)
WITH {
    eventName: e.name, 
    eventDescription: e.description,
    eventOwner: '${senderID}',
    subscribers: COLLECT({
        id: u.facebook_id,
        first_name: u.first_name
    })
} AS reminderData, e 
SET e.scheduled=TRUE
